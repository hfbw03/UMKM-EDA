FORMULA EXCEL MODIFIKASI DATASET MENTAH PROYEK SEGMENTASI UMKM

=IF(Q2<=0.01,
   RANDBETWEEN(40000000,60000000),
   G2
)

=IF(
  AND(G2>30000000, RAND()<0.5),
  RANDBETWEEN(8000000,18000000),
  G2
)

=IF(
   ISBLANK(G2),
   "",
   RANDBETWEEN(2000000,12000000)
   + G2*0.15
   + NORM.INV(RAND(),0,4000000)
)

=IF(
   ISBLANK(F2),
   "",
   IF(
      AND(F2>=G2, RAND()<0.65),
      G2 * RANDBETWEEN(30,80) / 100,
      F2
   )
)
